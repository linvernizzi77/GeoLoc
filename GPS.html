    <script>

        function getLocation() {

            if (navigator.geolocation) {

                navigator.geolocation.getCurrentPosition(function(position) {

                    var lat = position.coords.latitude;

                    var lon = position.coords.longitude;

                    document.getElementById("location").innerHTML = "Latitudine: " + lat + "<br>Longitudine: " + lon;


                    // Chiamata API Ninox per inviare la posizione

                    sendLocationToNinox(lat, lon);

                }, function(error) {

                    alert("Errore nel recupero della posizione: " + error.message);

                });

            } else {

                alert("La geolocalizzazione non è supportata da questo browser.");

            }

        }


        function sendLocationToNinox(lat, lon) {

            const apiUrl = "https://api.ninox.com/v1/teams/7bvSZXEyzhnuBgL3q/databases/Yughyi6xy9h99/records";

            const apiKey = MDU1N2FhYTEtYTIxNi01OGI0LWE1NTQtMjE1ZjMzZGVlODVl;

            

            const data = {

                "fields": {

                    "Latitude": lat,

                    "Longitude": lon,

                    "Timestamp": new Date().toISOString()

                }

            };


            fetch(apiUrl, {

                method: "POST",

                headers: {

                    "Content-Type": "application/json",

                    "Authorization": `Bearer ${apiKey}`

                },

                body: JSON.stringify(data)

            })

            .then(response => response.json())

            .then(data => {

                console.log("Posizione inviata a Ninox", data);

            })

            .catch(error => {

                console.error("Errore nell'invio della posizione a Ninox:", error);

            });

        }

    </script>
