    <h1>Acquisizione Posizione GPS</h1>

    <p>Premi il pulsante qui sotto per acquisire la tua posizione GPS:</p>

    <button id="getLocation">Ottieni Posizione GPS</button>


    <h2>Posizione Attuale:</h2>

    <p id="location">Nessuna posizione disponibile.</p>


    <script>

        // Funzione per ottenere la posizione GPS

        function getLocation() {

            if (navigator.geolocation) {

                navigator.geolocation.getCurrentPosition(showPosition, showError);

            } else {

                document.getElementById("location").innerHTML = "Geolocalizzazione non supportata dal browser.";

            }

        }


        // Funzione per mostrare la posizione

        function showPosition(position) {

            var latitude = position.coords.latitude;

            var longitude = position.coords.longitude;

            document.getElementById("location").innerHTML = "Latitudine: " + latitude + "<br>Longitudine: " + longitude;


            // Qui puoi aggiungere il codice per inviare i dati a Ninox tramite API, se necessario

            sendToNinox(latitude, longitude);

        }


        // Funzione per gestire gli errori

        function showError(error) {

            switch(error.code) {

                case error.PERMISSION_DENIED:

                    document.getElementById("location").innerHTML = "Permesso negato per ottenere la posizione.";

                    break;

                case error.POSITION_UNAVAILABLE:

                    document.getElementById("location").innerHTML = "Posizione non disponibile.";

                    break;

                case error.TIMEOUT:

                    document.getElementById("location").innerHTML = "Tempo di attesa scaduto per ottenere la posizione.";

                    break;

                case error.UNKNOWN_ERROR:

                    document.getElementById("location").innerHTML = "Errore sconosciuto durante l'acquisizione della posizione.";

                    break;

            }

        }


        // Funzione per inviare i dati a Ninox (questa parte è solo un esempio)

        function sendToNinox(latitude, longitude) {

            // In questo esempio, devi sostituire con l'API endpoint di Ninox o con il codice di integrazione

            // che stai utilizzando per inviare i dati a Ninox.

            console.log("Invio della posizione a Ninox: Latitudine - " + latitude + ", Longitudine - " + longitude);


            // Esempio di invio tramite fetch (modifica con l'URL e i parametri corretti di Ninox)

            fetch('https://effetre.ninoxdb.com//endpoint', {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json',

                    'Authorization': 'Bearer 8a46a100-03d4-11f0-ae07-3928b357d52c'

                },

                body: JSON.stringify({

                    latitudine: latitude,

                    longitudine: longitude

                })

            })

            .then(response => response.json())

            .then(data => console.log('Dati inviati a Ninox:', data))

            .catch(error => console.error('Errore nell\'invio a Ninox:', error));

        }


        // Aggiungi l'evento al pulsante

        document.getElementById("getLocation").addEventListener("click", getLocation);

    </script>
